// 1. Capturamos los elementos del DOM
const btnHackaton03 = document.getElementById('btn_hackaton03');
const inputHackaton03 = document.getElementById('input_hackaton03');

// 2. Agregamos el evento click al botón
btnHackaton03.addEventListener('click', () => {

    // 3. Convertimos el valor del input a número
    const valor = Number(inputHackaton03.value);

    // 4. Validamos que sea un número
    if (isNaN(valor) || inputHackaton03.value === '') {
        return alert('No es un número válido');
    }

    // 5. Verificamos si tiene 3 dígitos (100-999)
    if (valor >= 100 && valor <= 999) {
        alert('✅ El número tiene 3 dígitos');
    } else {
        alert('❌ El número NO tiene 3 dígitos');
    }
});
